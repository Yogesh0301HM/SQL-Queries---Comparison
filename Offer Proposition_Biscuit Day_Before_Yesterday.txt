SELECT count(*) AS Rec_Count, date(ocpb.last_modified_timestamp + INTERVAL '-1 hour') AS LastModifiedDate
from p1_cdw_dmbase..ft_onl_club_offer_proposition AS ocpb
where date(ocpb.last_modified_timestamp + INTERVAL '-1 hour') = current_date -2
GROUP BY date(ocpb.last_modified_timestamp + INTERVAL '-1 hour')
